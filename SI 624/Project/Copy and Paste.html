# Define a custom formatting function
def format_number(x):
    if x < 1e3:
        return x
    elif x < 1e6:
        return x / 1e3
    elif x < 1e9:
        return x / 1e6
    else:
        return x / 1e9

# Apply the custom formatting function to the describe result

import matplotlib.pyplot as plt
import seaborn as sns

# Filter the data for diabetic medication
diabetic_medication_spending = medicare[medicare['Gnrc_Name_Lower'].str.contains(medication_pattern, case=False, na=False) | medicare['Brnd_Name_Lower'].str.contains(medication_pattern, case=False, na=False)]
diabetic_medication_spending = diabetic_medication_spending[diabetic_medication_spending['Mftr_Name'] != 'Overall']

# Select the columns for years (e.g., Tot_Spndng_2017, Tot_Spndng_2018, etc.)
year_columns = ['Tot_Spndng_2017', 'Tot_Spndng_2018', 'Tot_Spndng_2019', 'Tot_Spndng_2020', 'Tot_Spndng_2021']

# Create lists to store years and corresponding total spending
years = []
total_spending = []

# Calculate total spending for each year and append to the lists
for year_column in year_columns:
    year = year_column.split('_')[2]
    total = diabetic_medication_spending[year_column].sum() / 1e9
    years.append(year)
    total_spending.append(total)
    print(f'Total spending in {year}: ${total:.2f}B')

# Create the bar plot
plt.figure(figsize=(12, 6))
plt.subplot(1, 2, 1)
sns.barplot(x=years, y=total_spending, palette='Set3')
plt.title('Diabetic Medication Spending Over the Years (Bar Plot)')
plt.xlabel('Year')
plt.ylabel('Total Spending (Billions)')

# Create the line plot
plt.subplot(1, 2, 2)
plt.plot(years, total_spending, marker='o', linestyle='-')
plt.title('Diabetic Medication Spending Over the Years (Line Plot)')
plt.xlabel('Year')
plt.ylabel('Total Spending (Billions)')

plt.tight_layout()
plt.show()
